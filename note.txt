#include <iostream>
#include <fstream>
#include <cmath>
#include "wave.h"

int main() {
    Wave::Wavedata wave = Wave::Wavedata::load("gomi.wav");
    if (wave.getlength() == 0) return -1;

    Wave::MonoAudio mono = Wave::MonoAudio::fromWavedata(wave);

    int N = 262144; // 2のべき乗
    if (mono.getlength() < N) {
        std::cout << "データが足りません" << std::endl;
        return -1;
    }

    Wave::cmpnum* fft_result = new Wave::cmpnum[N];
    
    Wave::FFT(mono.getdata() + (int) (44100 * 0), fft_result, N);

    std::ofstream ofs("fft_result.csv");
    if (ofs) {
        ofs << "Frequency,Magnitude" << std::endl;
        for (int i = 0; i < N / 2; i++) {
            double freq = (double)i * mono.getsamplerate() / N;
            double mag = std::sqrt(fft_result[i].real * fft_result[i].real + fft_result[i].imag * fft_result[i].imag);
            ofs << freq << "," << mag << std::endl;
        }
        ofs.close();
        std::cout << "Success: fft_result.csv" << std::endl;
    }

    delete[] fft_result;
    return 0;
}
